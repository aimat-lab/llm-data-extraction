\section{Approach}

\begin{frame}[c]{Criteria for Models}
    \Large
    \begin{enumerate}[<+(1)->]
    \item It is possible to get the full model weights.
    \item The selected models ought to be decently capable \glspl{causal}.
    \item Ceteris paribus, a smaller model is better.
    \end{enumerate}
\end{frame}

\begin{frame}[c]{Schema}
    \large
\code{schema.py}{schema}{The schema provided for the model to follow. Model output termination would happen after generation of a token for `\mintinline{python}{"}' for strings or `\texttt{,}' for numbers, or a number of other dedicated `end of generation' tokens, e.g. \texttt{<EOS>}.}
\end{frame}

\begin{frame}[c]{Prompt}
    \large
    \code{prompt.py}{prompt}{Prompt used to generate output. \mintinline{python}{"{output}"} delineates where the model provides an answer.}
\end{frame}

\begin{frame}[c]{Output}
    \large
    \code{example_output.py}{output}{Exemplary output based on the prompt and schema shown before.}
\end{frame}
